 from dataclasses import dataclass, asdict
 from typing import List


  class Skill:
    name: str
    level: str  # e.g., "Beginner", "Intermediate", "Expert"
    class Hobby:
    name: str
    description: str
    class Profile:
    skills: List[Skill]
    hobbies: List[Hobby]

    def to_json(self):
        return asdict(self)
	profile = Profile(
    skills=[
        Skill(name="Python", level="Expert"),
        Skill(name="Django", level="Intermediate"),
        Skill(name="SQL", level="Intermediate"),
    ],
    hobbies=[
        Hobby(name="Chess", description="Compete weekly in local club"),
        Hobby(name="Landscape Photography", description="Published in community gallery"),
    ]
)
import json
print(json.dumps(profile.to_json(), indent=2))
